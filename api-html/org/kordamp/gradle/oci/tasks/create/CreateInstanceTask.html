068 @CompileStatic
069 @TypeProviderFor(OCITask)
070 class CreateInstanceTask extends AbstractOCITask implements CompartmentIdAwareTrait,
071     SubnetIdAwareTrait,
072     InstanceNameAwareTrait,
073     ImageAwareTrait,
074     ShapeAwareTrait,
075     PublicKeyFileAwareTrait,
076     UserDataFileAwareTrait,
077     VerboseAwareTrait {
078     static final String TASK_DESCRIPTION = 'Creates an Instance.'
079 
080     private final Property<String> createdInstanceId = project.objects.property(String)
081 
082     String getCreatedInstanceId() {
083         return createdInstanceId.orNull
084     }

086     @Override
087     void doExecuteTask() {
088         validateCompartmentId()
089         validateSubnetId()
090         validateImage()
091         validateShape()
092         validatePublicKeyFile()
093 
094         ComputeClient computeClient = createComputeClient()
095 
096         Image _image = validateImage(computeClient, getCompartmentId())
097         Shape _shape = validateShape(computeClient, getCompartmentId())
098 
099         String publicKeyFile = getPublicKeyFile().text
100         String userDataFile = getUserDataFile()?.text
101         String kmsKeyId = ''
102 
103         VirtualNetworkClient vcnClient = createVirtualNetworkClient()
104         BlockstorageClient blockstorageClient = createBlockstorageClient()
105 
106         Subnet subnet = vcnClient.getSubnet(GetSubnetRequest.builder()
107             .subnetId(getSubnetId())
108             .build())
109             .subnet
110 
111         Instance instance = maybeCreateInstance(this,
112             computeClient,
113             vcnClient,
114             blockstorageClient,
115             getCompartmentId(),
116             getInstanceName(),
117             _image,
118             _shape,
119             subnet,
120             publicKeyFile,
121             userDataFile,
122             kmsKeyId,
123             isVerbose())
124         createdInstanceId.set(instance.id)
125     }

173     private static Instance doMaybeCreateInstance(OCITask owner,
174                                                   ComputeClient client,
175                                                   VirtualNetworkClient vcnClient,
176                                                   String compartmentId,
177                                                   String instanceName,
178                                                   String availabilityDomain,
179                                                   String imageId,
180                                                   String shape,
181                                                   String subnetId,
182                                                   String publicKeyFile,
183                                                   String userDataFile,
184                                                   String kmsKeyId,
185                                                   boolean verbose) {
186         // 1. Check if it exists
187         List<Instance> instances = client.listInstances(ListInstancesRequest.builder()
188             .compartmentId(compartmentId)
189             .availabilityDomain(availabilityDomain)
190             .displayName(instanceName)
191             .build())
192             .items
193 
194         if (!instances.empty) {
195             Instance instance = instances[0]
196             println("Instance '${instanceName}' already exists. id = ${owner.console.yellow(instance.id)}")
197             if (verbose) printInstance(owner, instance, 0)
198             return instances[0]
199         }
200 
201         Map<String, String> metadata = new HashMap<>('ssh_authorized_keys': publicKeyFile)
202         if (isNotBlank(userDataFile)) {
203             metadata.put("user_data", Base64.encodeBase64String(userDataFile.getBytes()))
204         }
205 
206         InstanceSourceViaImageDetails details =
207             (Strings.isNullOrEmpty(kmsKeyId))
208                 ? InstanceSourceViaImageDetails.builder()
209                 .imageId(imageId)
210                 .build()
211                 : InstanceSourceViaImageDetails.builder()
212                 .imageId(imageId)
213                 .kmsKeyId(kmsKeyId)
214                 .build()
215 
216         Instance instance = client.launchInstance(LaunchInstanceRequest.builder()
217             .launchInstanceDetails(LaunchInstanceDetails.builder()
218                 .availabilityDomain(availabilityDomain)
219                 .compartmentId(compartmentId)
220                 .displayName(instanceName)
221                 .metadata(metadata)
222                 .shape(shape)
223                 .sourceDetails(details)
224                 .createVnicDetails(CreateVnicDetails.builder()
225                     .subnetId(subnetId)
226                     .build())
227                 .agentConfig(LaunchInstanceAgentConfigDetails.builder()
228                     .isMonitoringDisabled(false)
229                     .build())
230                 .build())
231             .build())
232             .instance
233 
234         println("Waiting for Instance to be ${owner.state('Available')}")
235         client.waiters.forInstance(GetInstanceRequest.builder()
236             .instanceId(instance.id)
237             .build(),
238             Instance.LifecycleState.Running)
239             .execute()
240 
241         println("Instance '${instanceName}' has been provisioned. id = ${owner.console.yellow(instance.id)}")
242         if (verbose) printInstance(owner, instance, 0)
243 
244         Set<String> publicIps = getInstancePublicIp(owner,
245             client,
246             vcnClient,
247             compartmentId,
248             instance.id)
249 
250         owner.printCollection('Ip Addresses', publicIps, 0)
251 
252         instance
253     }

255     private static BootVolume maybeCreateBootVolume(OCITask owner,
256                                                     BlockstorageClient client,
257                                                     String compartmentId,
258                                                     String availabilityDomain,
259                                                     String imageId,
260                                                     String displayName,
261                                                     String kmsKeyId,
262                                                     boolean verbose) {
263         List<BootVolume> bootVolumes = client.listBootVolumes(ListBootVolumesRequest.builder()
264             .availabilityDomain(availabilityDomain)
265             .compartmentId(compartmentId)
266             .build())
267             .items
268 
269         String bootVolumeId = null
270         for (BootVolume bootVolume : bootVolumes) {
271             if (bootVolume.lifecycleState.equals(BootVolume.LifecycleState.Available)
272                 && bootVolume.imageId != null
273                 && bootVolume.imageId.equals(imageId)) {
274                 bootVolumeId = bootVolume.id
275                 break
276             }
277         }
278 
279         BootVolumeSourceDetails bootVolumeSourceDetails = BootVolumeSourceFromBootVolumeDetails.builder()
280             .id(bootVolumeId)
281             .build()
282 
283         CreateBootVolumeDetails.Builder details = CreateBootVolumeDetails.builder()
284             .availabilityDomain(availabilityDomain)
285             .compartmentId(compartmentId)
286             .displayName(displayName)
287             .sourceDetails(bootVolumeSourceDetails)
288 
289         if (isNotBlank(kmsKeyId)) {
290             details = details.kmsKeyId(kmsKeyId)
291         }
292 
293         BootVolume bootVolume = client.createBootVolume(CreateBootVolumeRequest.builder()
294             .createBootVolumeDetails(details.build())
295             .build())
296             .bootVolume
297 
298         println("Waiting for BootVolume to be ${owner.state('Available')}")
299         client.waiters.forBootVolume(
300             GetBootVolumeRequest.builder().bootVolumeId(bootVolumeId).build(),
301             BootVolume.LifecycleState.Available)
302             .execute()
303             .bootVolume
304 
305         println("BootVolume '${bootVolume.displayName}' has been provisioned. id = ${owner.console.yellow(bootVolume.id)}")
306         if (verbose) printBootVolume(owner, bootVolume, 0)
307         bootVolume
308     }
