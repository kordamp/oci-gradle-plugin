057 /**
058  * @author Andres Almiray
059  * @since 0.1.0
060  */
061 @CompileStatic
062 @TypeProviderFor(OCITask)
063 class SetupInstanceTask extends AbstractOCITask implements CompartmentIdAwareTrait,
064     InstanceNameAwareTrait,
065     ImageAwareTrait,
066     ShapeAwareTrait,
067     PublicKeyFileAwareTrait,
068     UserDataFileAwareTrait,
069     VerboseAwareTrait {
070     static final String TASK_DESCRIPTION = 'Setups an Instance with Vcn, InternetGateway, Subnets, InstanceConsoleConnection, and Volume.'
071 
072     private final Property<String> createdInstanceId = project.objects.property(String)
073     private final Provider<RegularFile> output
074 
075     SetupInstanceTask() {
076         output = getResolvedInstanceName().map(new Transformer<RegularFile, String>() {
077             @Override
078             RegularFile transform(String s) {
079                 return project.layout.buildDirectory.file("oci/instance/${s}.properties").get()
080             }
081         })
082     }
083 
084     @Internal
085     Property<String> getCreatedInstanceId() {
086         this.@createdInstanceId
087     }
088 
089     @OutputFile
090     Provider<RegularFile> getOutput() {
091         this.@output
092     }
093 
094     String outputProperty(String key) {
095         Properties props = new Properties()
096         props.load(output.get().asFile.newInputStream())
097         props.get(key)
098     }
099 
100     @Override
101     protected void doExecuteTask() {
102         validateCompartmentId()
103         validateImage()
104         validateShape()
105         validatePublicKeyFile()
106 
107         output.get().asFile.parentFile.mkdirs()
108 
109         Properties props = new Properties()
110         props.put('compartment.id', getResolvedCompartmentId().get())
111 
112         ComputeClient computeClient = createComputeClient()
113 
114         Image _image = validateImage(computeClient, getResolvedCompartmentId().get())
115         Shape _shape = validateShape(computeClient, getResolvedCompartmentId().get())
116 
117         String networkCidrBlock = '10.0.0.0/16'
118         File publicKeyFile = getResolvedPublicKeyFile().get().asFile
119         File userDataFile = getResolvedUserDataFile().get()?.asFile
120         String vcnDisplayName = getResolvedInstanceName().get() + '-vcn'
121         String dnsLabel = getResolvedInstanceName().get()
122         String internetGatewayDisplayName = getResolvedInstanceName().get() + '-internet-gateway'
123         String kmsKeyId = ''
124 
125         IdentityClient identityClient = createIdentityClient()
126         VirtualNetworkClient vcnClient = createVirtualNetworkClient()
127         BlockstorageClient blockstorageClient = createBlockstorageClient()
128 
129         Vcn vcn = maybeCreateVcn(this,
130             vcnClient,
131             getResolvedCompartmentId().get(),
132             vcnDisplayName,
133             dnsLabel,
134             networkCidrBlock,
135             true,
136             getResolvedVerbose().get())
137         props.put('vcn.id', vcn.id)
138         props.put('vcn.name', vcn.displayName)
139         props.put('vcn.security-list.id', vcn.defaultSecurityListId)
140         props.put('vcn.route-table.id', vcn.defaultRouteTableId)
141 
142         InternetGateway internetGateway = maybeCreateInternetGateway(this,
143             vcnClient,
144             getResolvedCompartmentId().get(),
145             vcn.id,
146             internetGatewayDisplayName,
147             true,
148             getResolvedVerbose().get())
149         props.put('internet-gateway.id', internetGateway.id)
150 
151         Subnet subnet = null
152         int subnetIndex = 0
153         // create a Subnet per AvailabilityDomain
154         List<AvailabilityDomain> availabilityDomains = identityClient.listAvailabilityDomains(ListAvailabilityDomainsRequest.builder()
155             .compartmentId(getResolvedCompartmentId().get())
156             .build()).items
157         props.put('vcn.subnets', availabilityDomains.size().toString())
158         for (AvailabilityDomain domain : availabilityDomains) {
159             String subnetDnsLabel = 'sub' + HashUtil.sha1(vcn.id.bytes).asHexString()[0..8] + (subnetIndex.toString().padLeft(3, '0'))
160 
161             Subnet s = maybeCreateSubnet(this,
162                 vcnClient,
163                 getResolvedCompartmentId().get(),
164                 vcn.id,
165                 subnetDnsLabel,
166                 domain.name,
167                 'Subnet ' + domain.name,
168                 "10.0.${subnetIndex}.0/24".toString(),
169                 true,
170                 getResolvedVerbose().get())
171             props.put("subnet.${subnetIndex}.id".toString(), s.id)
172             props.put("subnet.${subnetIndex}.name".toString(), s.displayName)
173 
174             // save the first one
175             if (subnet == null) subnet = s
176             subnetIndex++
177         }
178 
179         Instance instance = maybeCreateInstance(this,
180             computeClient,
181             vcnClient,
182             blockstorageClient,
183             getResolvedCompartmentId().get(),
184             getResolvedInstanceName().get(),
185             _image,
186             _shape,
187             subnet,
188             publicKeyFile,
189             userDataFile,
190             kmsKeyId,
191             true)
192         createdInstanceId.set(instance.id)
193         props.put('instance.id', instance.id)
194         props.put('instance.name', instance.displayName)
195 
196         Set<String> publicIps = getInstancePublicIp(this,
197             computeClient,
198             vcnClient,
199             getResolvedCompartmentId().get(),
200             instance.id)
201 
202         props.put('instance.public-ips', publicIps.size().toString())
203         int publicIpIndex = 0
204         for (String publicIp : publicIps) {
205             props.put("instance.public-ip.${publicIpIndex++}".toString(), publicIp)
206         }
207 
208         props.store(new FileWriter(getOutput().get().asFile), '')
209         println("Result stored at ${console.yellow(getOutput().get().asFile.absolutePath)}")
210     }
211 }
